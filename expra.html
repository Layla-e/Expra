#TRIAL LOOP
for stim_text, correct_key in trials:

    #Fixation-cross
    fixation.draw()
    win.flip()
    core.wait(0.5)

    #Stimulus
    stim = visual.TextStim(win, text=stim_text, color="white")
    stim.draw()
    win.flip()

    response = event.waitKeys(keyList=["s", "l", "escape"])[0]

    #QUICK EXIT THAT FINALLY WORKS OML
    if response == "escape":
        win.close()
        core.quit()

    #Feedback
    if response == correct_key:
        feedback.text = "correct"
        feedback.draw()
        win.flip()
        core.wait(0.5)
    else:
        feedback.text = "false"
        feedback.draw()
        win.flip()
        core.wait(1.5)

    #ITI
    blank.draw()
    win.flip()
    core.wait(0.5)
